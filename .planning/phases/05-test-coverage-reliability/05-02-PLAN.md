```python
class CrawlerEventListener(ABC):
    @abstractmethod def on_crawl_started(self, run_id: int, target_package: str) -> None: ...
    @abstractmethod def on_step_started(self, run_id: int, step_number: int) -> None: ...
    @abstractmethod def on_screenshot_captured(self, run_id: int, step_number: int, screenshot_path: str) -> None: ...
    @abstractmethod def on_ai_request_sent(self, run_id: int, step_number: int, request_data: Dict[str, Any]) -> None: ...
    @abstractmethod def on_ai_response_received(self, run_id: int, step_number: int, response_data: Dict[str, Any]) -> None: ...
    @abstractmethod def on_action_executed(self, run_id: int, step_number: int, action_index: int, result: ActionResult) -> None: ...
    @abstractmethod def on_step_completed(self, run_id: int, step_number: int, actions_count: int, duration_ms: float) -> None: ...
    @abstractmethod def on_crawl_completed(self, run_id, total_steps, total_duration_ms, reason, ocr_avg_ms=0.0) -> None: ...
    @abstractmethod def on_error(self, run_id: int, step_number: Optional[int], error: Exception) -> None: ...
    @abstractmethod def on_state_changed(self, run_id: int, old_state: str, new_state: str) -> None: ...
    @abstractmethod def on_screen_processed(self, run_id, step_number, screen_id, is_new, visit_count, total_screens) -> None: ...
    @abstractmethod def on_debug_log(self, run_id: int, step_number: int, message: str) -> None: ...
    @abstractmethod def on_ocr_completed(self, run_id: int, step_number: int, duration_ms: float, element_count: int) -> None: ...
    @abstractmethod def on_screenshot_timing(self, run_id: int, step_number: int, duration_ms: float) -> None: ...
    def on_recovery_started(self, run_id, step_number, attempt_number) -> None: ...
    def on_recovery_completed(self, run_id, step_number, success, duration_ms) -> None: ...
    def on_recovery_exhausted(self, run_id, step_number, attempts, message) -> None: ...
    def on_step_phase_transition(self, run_id, step_number, from_phase, to_phase, duration_ms) -> None: ...
```
